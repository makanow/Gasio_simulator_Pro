// 修正: 日本語ヘッダー（文字列不問）対応のマスターアップロード関数 async function uploadMaster() { const file = document.getElementById('file_master').files[0]; const statusEl = document.getElementById('status_master'); if (!file) return; statusEl.innerText = '解析中...'; statusEl.className = 'status loading'; const reader = new FileReader(); reader.onload = function(e) { const buffer = new Uint8Array(e.target.result); const unicodeString = Encoding.convert(buffer, { to: 'UNICODE', type: 'string' }); Papa.parse(unicodeString, { header: false, // ★列の順番で読み込む（文字列不問） skipEmptyLines: true, complete: async function(results) { let rawData = results.data; // ★1行目がヘッダー（数字以外）の場合は自動スキップ if (rawData.length > 0 && isNaN(parseInt(rawData[0][0]))) { rawData.shift(); } // 0〜5番目の列順でデータを取得 const insertData = rawData.map(row => ({ user_id: currentUser.id, plan_id: parseInt(cleanNum(row[0])), plan_name: String(row[1] || "").trim(), min_usage: cleanNum(row[2]), max_usage: cleanNum(row[3]), base_fee: cleanNum(row[4]), unit_price: cleanNum(row[5]) })).filter(d => !isNaN(d.plan_id)); const { error } = await client.from('price_masters').insert(insertData); if (error) { statusEl.innerText = 'エラー: ' + error.message; statusEl.className = 'status error'; } else { statusEl.innerText = '✓ アップロード完了'; statusEl.className = 'status success'; updateCounts(); loadEditableMaster(); } } }); }; reader.readAsArrayBuffer(file); }
